270 std::string LexemeLibrary::apply_transform(const Lexeme& toperator, const std::string& a, const std::string& b, bool in_parenthesis)
271 {
272  const bool space_wrapping = true;
273  std::string space = space_wrapping ? " " : "";
274  //----------------------------------------------------
275  auto replace_op_with_tag = [&](const std::string& op, const std::string& a, const std::string& b) {
276  std::stringstream ss;
277  ss << a << space << tags[op] << space << b;
278  return ss.str();
279  };
280  //----------------------------------------------------
281  std::string result;
282 
283  switch(toperator.type())
284  {
285  case LexemeLibrary::operation:
286  if (toperator.lexeme() == "/")
287  {
288  std::stringstream ss;
289  ss << tags["/"] << "{" << a << "}" << "{" << b << "}";
290  result = ss.str();
291  }
292  else if (tags.find(toperator.lexeme()) != tags.end())
293  {
294  result = replace_op_with_tag(toperator.lexeme(), a, b);
295  }
296  else
297  {
298  result = a + apply_default_transform(space + toperator.lexeme() + space) + b;
299  }
300 
301  if(in_parenthesis)
302  {
303  result = tags["("] + space + result + space + tags[")"];
304  }
305  break;
306  case LexemeLibrary::function:
307 
308  if (toperator.lexeme() == "sqrt")
309  {
310  result = R"!(\sqrt{)!" + (b.empty() ? a : b) + "}";
311  }
312  else
313  {
314  std::string x = b.empty() ? a : b;
315 
316  if(!starts_with(x, tags["("]) && !ends_with(x, tags[")"]))
317  {
318  result = toperator.lexeme() + space + tags["("] + x + tags[")"];
319  }
320  else
321  {
322  result = toperator.lexeme() + space + x;
323  }
324  }
325  break;
326  case LexemeLibrary::index:
327  result = replace_op_with_tag(toperator.lexeme(), a, b);
328  break;
329  default:
330  GLogger::instance().logError("Unsupported type:", toperator.type(), " of lexeme:", toperator.lexeme());
331  break;
332  }
333 
334 
335  return result;
336 }
