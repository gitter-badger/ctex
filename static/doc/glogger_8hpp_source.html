339 void GLogger::write_log_message(Level level, std::string& message)
340 {
341  if(trim_messages_)
342  trim(message);
343 
344  if (output_ != Off && !(skip_empty_msgs_ && message.empty()) )
345  {
346  std::string snow = current_time();
347  // write to log file
348  if ((output_ == File || output_ == Both) && level >= min_level_file_)
349  {
350  if (fout_.good())
351  {
352  fout_ << level_name(level) << separator_ << "[" << snow << "]" << separator_ << message << std::endl;
353  }
354  }
355  // write to console
356  if ((output_ == Console || output_ == Both) && level >= min_level_console_)
357  {
358  if (level == Level::Error)
359  {
360  std::cerr << level_name(level) << separator_ << message << std::endl;
361  }
362  else
363  {
364  std::cout << level_name(level) << separator_ << message << std::endl;
365  }
366  }
367  }
368  // recording
369  if (record_enabled_ && level >= min_level_console_)
370  buffer_record_ << message << std::endl;
371 }
